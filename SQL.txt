ELECT TOP(50) C.Cpf, 
       DA.Acionado
  FROM [MOVEDB].[dbo].Contrato C --> PK
  JOIN [CAMPANHAS]..Dados_Pessoais DP -- FK
    ON DP.Cpf = C.Cpf 
--       FK      PK
  JOIN [CAMPANHAS]..Dados_Beneficio DB -->FK
    ON DB.Cpf = DP.Cpf
--      FK       PK
  JOIN [CAMPANHAS]..Dados_Acionamento DA --> FK
    ON DB.Beneficio = DA.Beneficio
--       PK             FK
